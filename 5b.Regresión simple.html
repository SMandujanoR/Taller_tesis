datos suponiendo que fuimos a 50 charcas

set.seed(34); n <- 50; sigma <- 2; b0 <- 2; b1 <- 0.3 
x <- runif(n, 10, 30)
yhat <- b0 + b1*x
y <- rnorm(n, yhat, sd = sigma)

graficamos nuevos datos:

mod <-  lm(y~x)
par(mfrow = c(1,1))
